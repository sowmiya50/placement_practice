recursion function

function to find factorial

#include<stdio.h>
long long int factorial(int num)
{

    if(num==0 || num==1)
        return 1;
    return num*factorial(num-1);
}
int main()
{
    int num;
    scanf("%d",&num);
    printf("%lld",factorial(num));
}
output
5
120

reversing the digits using recursion function

#include<stdio.h>
 int digits(int num)
{
    if(num==0)
        return 0;
    printf("%d",num%10);
    digits(num/10);
}
int main()
{
    int num;
    scanf("%d",&num);
    digits(num);
}

output
165
561

Fibonacci series

#include<stdio.h>
 int fibo(int num)
{
  if(num==0)
       return 0;
  else if(num==1 || num==2)
        return 1;
  return fibo(num-1)+ fibo(num-2);
}
int main()
{
    int num;
    scanf("%d",&num);
    printf("%d",fibo(num));

}
 output
3
2
