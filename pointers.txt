#include<stdio.h>
int main()
{
int size,even=0,odd=0;
    scanf("%d", &size);
    int arr[size];
    for(int itr=0;itr<size;itr++)
        scanf("%d",&arr[itr]);
    for(int itr=0;itr<size;itr++)
    {
        if(arr[itr]%2==0)
           even++;
         else
            odd++;
    }
    printf("even count= %d\n",even);
    printf("odd count= %d\n",odd);
}
